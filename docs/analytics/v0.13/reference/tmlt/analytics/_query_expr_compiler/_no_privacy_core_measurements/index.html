class NPSum(input_domain, input_metric, measure_column)#

Bases: tmlt.core.measurements.base.Measurement

Returns the sum of a single numeric column in a spark DataFrame.

Warning

This measurement does not satisfy differential privacy (has an infinite d_out).

Example

>>> # Example input
>>> print_sdf(spark_dataframe)
    A  X
0  a1  2
1  a1  3
2  a2 -1
3  a2  5
>>> # Create the measurement
>>> sum_X = NPSum(
...     input_domain=SparkDataFrameDomain(
...         {
...             "A": SparkStringColumnDescriptor(),
...             "X": SparkIntegerColumnDescriptor(),
...         },
...     ),
...     input_metric=SymmetricDifference(),
...     measure_column="X",
... )
>>> # Apply measurement to data
>>> sum_X(spark_dataframe)
9
>>> sum_X.privacy_function(1)

class NPSumGrouped(input_domain, input_metric, measure_column, sum_column=None, lower=None, upper=None)#

Bases: tmlt.core.measurements.base.Measurement

Computes the sum of a column for each group in a GroupedDataFrame.

Warning

This measurement does not satisfy differential privacy (has an infinite d_out).

Example

>>> # Example input
>>> print_sdf(spark_dataframe)
    A  X
0  a1  2
1  a1  3
2  a2 -1
3  a2  5
>>> # Specify group keys
>>> group_keys = spark.createDataFrame(
...     [("a0",), ("a2",)],
...     schema=["A"],
... )
>>> # Note that we omit the key 'a1' even though it
>>> # exists in the spark dataframe and include 'a0'.
>>> # Create the measurement
>>> sum_X_by_A = NPSumGrouped(
...     input_domain=SparkGroupedDataFrameDomain(
...         schema={
...             "A": SparkStringColumnDescriptor(),
...             "X": SparkIntegerColumnDescriptor(),
...         },
...         groupby_columns=["A"],
...     ),
...     input_metric=SumOf(SymmetricDifference()),
...     measure_column="X",
... )
>>> # Create GroupedDataFrame
>>> grouped_dataframe = GroupedDataFrame(
...     dataframe=spark_dataframe,
...     group_keys=group_keys,
... )
>>> # Apply measurement to data
>>> print_sdf(sum_X_by_A(grouped_dataframe))
    A  sum(X)
0  a0       0
1  a2       4

>>> sum_X_by_A.privacy_function(1)
